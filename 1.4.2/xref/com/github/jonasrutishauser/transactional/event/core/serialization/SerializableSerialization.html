3   import java.io.ByteArrayInputStream;
4   import java.io.ByteArrayOutputStream;
5   import java.io.IOException;
6   import java.io.ObjectInputStream;
7   import java.io.ObjectOutputStream;
8   import java.io.Serializable;
9   import java.nio.charset.StandardCharsets;
10  import java.util.Base64;
11  
12  import javax.annotation.Priority;
13  import javax.enterprise.context.Dependent;
14  
15  import com.github.jonasrutishauser.transactional.event.api.serialization.GenericSerialization;
16  
17  @Dependent
18  @Priority(1000)
19  public class SerializableSerialization implements GenericSerialization {
20  
21      @Override
22      public boolean accepts(Class<?> type) {
23          return Serializable.class.isAssignableFrom(type);
24      }
25  
26      @Override
27      public String serialize(Object event) {
28          ByteArrayOutputStream bout = new ByteArrayOutputStream();
29          try (ObjectOutputStream out = new ObjectOutputStream(Base64.getEncoder().wrap(bout))) {
30              out.writeObject(event);
31          } catch (IOException e) {
32              throw new IllegalStateException(e);
33          }
34          return new String(bout.toByteArray(), StandardCharsets.UTF_8);
35      }
36  
37      @Override
38      public <T> T deserialize(String event, Class<T> type) {
39          try (ObjectInputStream in = new ObjectInputStream(
40                  Base64.getDecoder().wrap(new ByteArrayInputStream(event.getBytes(StandardCharsets.UTF_8))))) {
41              return type.cast(in.readObject());
42          } catch (IOException | ClassNotFoundException e) {
43              throw new IllegalStateException(e);
44          }
45      }
46  
47  }

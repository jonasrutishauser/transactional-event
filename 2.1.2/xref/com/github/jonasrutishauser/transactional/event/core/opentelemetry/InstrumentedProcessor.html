23  import io.opentelemetry.api.trace.SpanBuilder;
24  import io.opentelemetry.api.trace.Tracer;
25  import io.opentelemetry.context.Context;
26  import io.opentelemetry.context.Scope;
27  import io.opentelemetry.context.propagation.TextMapGetter;
28  import io.opentelemetry.context.propagation.TextMapPropagator;
29  
30  @Decorator
31  @Priority(LIBRARY_BEFORE)
32  public class InstrumentedProcessor implements ContextualProcessor {
33  
34      private final ContextualProcessor delegate;
35  
36      private final Tracer tracer;
37      private final TextMapPropagator propagator;
38      private final String lockOwnerId;
39      private final TextMapGetter<Properties> getter = new TextMapGetter<Properties>() {
40          @Override
41          public String get(Properties carrier, String key) {
42              return carrier.getProperty(key);
43          }
44  
45          @Override
46          public Iterable<String> keys(Properties carrier) {
47              return carrier.stringPropertyNames();
48          }
49      };
50  
51      InstrumentedProcessor() {
52          this(null, null, null, null);
53      }
54  
55      @Inject
56      InstrumentedProcessor(@Delegate @Any ContextualProcessor delegate, @Events Tracer tracer,
57              @Events TextMapPropagator propagator, @Named("lockOwner.id") String lockOwnerId) {
58          this.delegate = delegate;
59          this.tracer = tracer;
60          this.propagator = propagator;
61          this.lockOwnerId = lockOwnerId;
62      }
63  
64      @Override
65      public void process(String id, String type, Properties context, String payload, Handler handler) {
66          Context extractedContext = propagator.extract(Context.current(), context, getter);
67          SpanBuilder spanBuilder = tracer.spanBuilder(type + " process") //
68                  .setSpanKind(CONSUMER) //
69                  .setAttribute("messaging.system", "transactional-event") //
70                  .setAttribute("messaging.destination", type) //
71                  .setAttribute("messaging.message_id", id) //
72                  .setAttribute("messaging.message_payload_size_bytes", payload.getBytes(UTF_8).length) //
73                  .setAttribute("messaging.operation", "process") //
74                  .setAttribute("messaging.consumer_id", lockOwnerId);
75          if (extractedContext != context) {
76              spanBuilder.addLink(Span.fromContext(extractedContext).getSpanContext());
77          }
78          Span span = spanBuilder.startSpan();
79          try (Scope unused = span.makeCurrent()) {
80              delegate.process(id, type, context, payload, handler);
81          } catch (RuntimeException e) {
82              span.setStatus(ERROR, "Processing failed");
83              span.recordException(e, Attributes.builder().put("exception.escaped", true).build());
84              throw e;
85          } finally {
86              span.end();
87          }
88      }
89  
90  }

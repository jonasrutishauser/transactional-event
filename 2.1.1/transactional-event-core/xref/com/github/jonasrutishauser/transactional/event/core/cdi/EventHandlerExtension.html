43      private final Set<ParameterizedType> requiredEventDeserializers = new HashSet<>();
44      private final Set<Class<?>> genericSerializationEventTypes = new HashSet<>();

46      private final Map<ParameterizedType, Class<? extends Handler>> handlerClass = new HashMap<>();

48      @Override
49      public Optional<Class<? extends Handler>> getHandlerClassWithImplicitType(EventTypeResolver typeResolver,
50              String type) {
51          for (Entry<ParameterizedType, Class<? extends Handler>> handlerClassEntry : handlerClass.entrySet()) {
52              if (type.equals(typeResolver.resolve((Class<?>) handlerClassEntry.getKey().getActualTypeArguments()[0]))) {
53                  return Optional.of(handlerClassEntry.getValue());
54              }
55          }
56          return Optional.empty();
57      }

59      <T extends Handler> void processHandlers(@Observes @Priority(LIBRARY_AFTER) ProcessBean<T> beanEvent) {
60          if (!beanEvent.getAnnotated().isAnnotationPresent(EventHandler.class)) {
61              beanEvent.addDefinitionError(
62                      new IllegalStateException("EventHandler annotation is missing on bean " + beanEvent.getBean()));
63          } else {
64              EventHandler annotation = beanEvent.getAnnotated().getAnnotation(EventHandler.class);
65              Optional<ParameterizedType> abstractHandlerType = getAbstractHandlerType(beanEvent.getBean().getTypes());
66              if (EventHandler.ABSTRACT_HANDLER_TYPE.equals(annotation.eventType())) {
67                  if (!abstractHandlerType.isPresent()) {
68                      beanEvent.addDefinitionError(new IllegalStateException("AbstractHandler type is missing on bean "
69                              + beanEvent.getBean() + " with implicit event type"));
70                  } else if (!beanEvent.getBean().getTypes().contains(beanEvent.getBean().getBeanClass())) {
71                      beanEvent.addDefinitionError(
72                              new IllegalStateException(beanEvent.getBean().getBeanClass().getSimpleName()
73                                      + " type is missing on bean " + beanEvent.getBean() + " with implicit event type"));
74                  } else {
75                      handlerClass.put(abstractHandlerType.get(),
76                              beanEvent.getBean().getBeanClass().asSubclass(Handler.class));
77                  }
78              }
79  
80          }
81      }

83      <X extends EventDeserializer<?>> void processEventDeserializerInjections(
84              @Observes ProcessInjectionPoint<?, X> event) {
85          Type type = event.getInjectionPoint().getType();
86          if (type instanceof ParameterizedType
87                  && EventDeserializer.class.equals(((ParameterizedType) type).getRawType())) {
88              requiredEventDeserializers.add((ParameterizedType) type);
89          }
90      }

92      void addMissingEventDeserializers(@Observes @Priority(LIBRARY_AFTER) AfterBeanDiscovery event,
93              BeanManager beanManager) {
94          for (ParameterizedType type : requiredEventDeserializers) {
95              try {
96                  if (beanManager.resolve(beanManager.getBeans(type)) == null) {
97                      Type eventType = type.getActualTypeArguments()[0];
98                      Class<?> eventClass = eventType instanceof Class ? (Class<?>) eventType
99                              : (Class<?>) ((ParameterizedType) eventType).getRawType();
100                     genericSerializationEventTypes.add(eventClass);
101                     event.addBean() //
102                             .addType(type) //
103                             .addType(ExtendedEventDeserializer.class) //
104                             .scope(ApplicationScoped.class) //
105                             .qualifiers(Default.Literal.INSTANCE) //
106                             .produceWith(instance -> createDefaultEventDeserializer(
107                                     instance.select(GenericSerialization.class), eventClass));
108                 }
109             } catch (AmbiguousResolutionException e) {
110                 // ignore, will be reported later
111             }
112         }
113     }

115     void verifyGenericSerializationEventTypes(@Observes AfterDeploymentValidation event, BeanManager beanManager) {
116         List<GenericSerialization> serializations = new ArrayList<>();
117         Instance<GenericSerialization> instance = beanManager.createInstance().select(GenericSerialization.class,
118                 Literal.INSTANCE);
119         instance.forEach(serializations::add);
120         for (Class<?> eventType : genericSerializationEventTypes) {
121             if (serializations.stream().noneMatch(s -> s.accepts(eventType))) {
122                 event.addDeploymentProblem(
123                         new UnsatisfiedResolutionException("No GenericSerialization found for " + eventType));
124             }
125         }
126         serializations.forEach(instance::destroy);
127     }

129     public static <T> DefaultEventDeserializer<T> createDefaultEventDeserializer(Instance<GenericSerialization> instance,
130             Class<T> type) {
131         List<GenericSerialization> serializations = new ArrayList<>();
132         instance.forEach(serializations::add);
133         sort(serializations, comparing(GenericSerialization::priority));
134         DefaultEventDeserializer<T> result = null;
135         for (GenericSerialization serialization : serializations) {
136             if (result == null && serialization.accepts(type)) {
137                 result = new DefaultEventDeserializer<>(type, serialization);
138             } else {
139                 instance.destroy(serialization);
140             }
141         }
142         if (result == null) {
143             throw new UnsatisfiedResolutionException("No GenericSerialization found for " + type);
144         }
145         return result;
146     }

148     private Optional<ParameterizedType> getAbstractHandlerType(Set<Type> types) {
149         return types.stream() //
150                 .filter(ParameterizedType.class::isInstance) //
151                 .map(ParameterizedType.class::cast) //
152                 .filter(type -> AbstractHandler.class.equals(type.getRawType())) //
153                 .findAny();
154     }

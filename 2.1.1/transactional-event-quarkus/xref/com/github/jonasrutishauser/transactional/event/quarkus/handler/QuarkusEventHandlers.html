18  import jakarta.inject.Inject;

21  class QuarkusEventHandlers implements EventHandlers {
22  
23      private final Map<Class<?>, Class<? extends Handler>> handlerClass;
24      private final Invoker<?, ?> startupMethod;
25  
26      QuarkusEventHandlers(Map<Class<?>, Class<? extends Handler>> handlerClass, Invoker<?, ?> startupMethod) {
27          this.handlerClass = handlerClass;
28          this.startupMethod = startupMethod;
29      }
30  
31      @Override
32      public Optional<Class<? extends Handler>> getHandlerClassWithImplicitType(EventTypeResolver typeResolver,
33              String type) {
34          for (Entry<Class<?>, Class<? extends Handler>> handlerClassEntry : handlerClass.entrySet()) {
35              if (type.equals(typeResolver.resolve(handlerClassEntry.getKey()))) {
36                  return Optional.of(handlerClassEntry.getValue());
37              }
38          }
39          return Optional.empty();
40      }
41  
42      void startup(StartupEvent event) throws Exception {
43          if (startupMethod != null) {
44              startupMethod.invoke(null, new Object[] {event});
45          }
46      }
47  
48      @Singleton
49      static class Startup {
50          private final QuarkusEventHandlers handlers;
51          
52          @Inject
53          Startup(QuarkusEventHandlers handlers) {
54              this.handlers = handlers;
55          }
56          
57          void startup(@Observes StartupEvent event) throws Exception {
58              handlers.startup(event);
59          }
60      }
61  
62      static class Creator implements SyntheticBeanCreator<EventHandlers> {
63  
64          @Override
65          public QuarkusEventHandlers create(Instance<Object> lookup, Parameters params) {
66              Class<?>[] types = params.get("types", Class[].class);
67              Class<?>[] beans = params.get("beans", Class[].class);
68              Map<Class<?>, Class<? extends Handler>> handlerClass = new HashMap<>();
69              for (int i = 0; i < types.length; i++) {
70                  handlerClass.put(types[i], beans[i].asSubclass(Handler.class));
71              }
72              return new QuarkusEventHandlers(handlerClass, params.get("startup", Invoker.class));
73          }
74  
75      }
76  
77  }

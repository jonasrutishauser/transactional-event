3   import java.util.Collection;
4   import java.util.HashSet;
5   import java.util.Set;
6   import java.util.concurrent.CopyOnWriteArraySet;
7   
8   import jakarta.enterprise.context.ApplicationScoped;
9   
10  @ApplicationScoped
11  public class Messages {
12  
13      private final Set<String> items = new CopyOnWriteArraySet<>();
14      private final Set<String> failures = new HashSet<>();
15  
16      protected void add(String message) {
17          items.add(message);
18      }
19  
20      public Collection<String> get() {
21          return items;
22      }
23  
24      protected boolean addFailure(String message) {
25          if (!failures.add(message)) {
26              failures.remove(message);
27              return false;
28          }
29          return true;
30      }
31  }

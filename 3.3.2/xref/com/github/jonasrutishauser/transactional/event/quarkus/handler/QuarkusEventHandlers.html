17  class QuarkusEventHandlers implements EventHandlers {
18  
19      private final Set<Class<?>> handledTypes;
20  
21      QuarkusEventHandlers(Set<Class<?>> handledTypes) {
22          this.handledTypes = handledTypes;
23      }
24  
25      @Override
26      public Annotation getHandlerQualifier(EventTypeResolver typeResolver, String type) {
27          for (Class<?> handledType : handledTypes) {
28              if (type.equals(typeResolver.resolve(handledType))) {
29                  return TypedEventHandler.Literal.of(handledType);
30              }
31          }
32          return EventHandlerLiteral.of(type);
33      }
34  
35      static class Creator implements SyntheticBeanCreator<EventHandlers> {
36          @Override
37          public QuarkusEventHandlers create(Instance<Object> lookup, Parameters params) {
38              Class<?>[] types = params.get("types", Class[].class);
39              Set<Class<?>> typeSet = ConcurrentHashMap.newKeySet();
40              typeSet.addAll(Arrays.asList(types));
41              return new QuarkusEventHandlers(typeSet);
42          }
43      }

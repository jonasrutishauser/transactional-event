1   package com.github.jonasrutishauser.transactional.event.core.opentelemetry;
2   
3   import javax.enterprise.context.Dependent;
4   import javax.enterprise.inject.Produces;
5   
6   import io.opentelemetry.api.GlobalOpenTelemetry;
7   import io.opentelemetry.api.OpenTelemetry;
8   import io.opentelemetry.api.trace.Tracer;
9   import io.opentelemetry.context.propagation.TextMapPropagator;
10  
11  @Dependent
12  class Instrumenter {
13      private Instrumenter() {
14      }
15  
16      @Produces
17      static TextMapPropagator getPropagator(OpenTelemetry openTelemetry) {
18          return openTelemetry.getPropagators().getTextMapPropagator();
19      }
20  
21      @Produces
22      static Tracer getTracer(OpenTelemetry openTelemetry) {
23          return openTelemetry.getTracer("transactional-event",
24                  Instrumenter.class.getPackage().getImplementationVersion());
25      }
26  
27      @Produces
28      static OpenTelemetry getGlobalOpenTelemetry() {
29          return GlobalOpenTelemetry.get();
30      }
31  }

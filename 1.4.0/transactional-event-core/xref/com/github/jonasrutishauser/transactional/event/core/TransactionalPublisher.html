10  import javax.enterprise.context.Dependent;
11  import javax.enterprise.event.Event;
12  import javax.inject.Inject;
13  import javax.transaction.Transactional;
14  
15  import org.apache.logging.log4j.LogManager;
16  import org.apache.logging.log4j.Logger;
17  
18  import com.github.jonasrutishauser.transactional.event.api.context.ContextualPublisher;
19  import com.github.jonasrutishauser.transactional.event.api.monitoring.PublishingEvent;
20  
21  @Dependent
22  class TransactionalPublisher implements ContextualPublisher {
23  
24      private static final Logger LOGGER = LogManager.getLogger();
25  
26      private final PublishedEvents publishedEvents;
27      private final Event<PublishingEvent> publishingEvent;
28  
29      TransactionalPublisher() {
30          this(null, null);
31      }
32  
33      @Inject
34      TransactionalPublisher(PublishedEvents publishedEvents, Event<PublishingEvent> publishingEvent) {
35          this.publishedEvents = publishedEvents;
36          this.publishingEvent = publishingEvent;
37      }
38  
39      @Override
40      @Transactional(MANDATORY)
41      public void publish(String id, String type, Properties context, String payload) {
42          PendingEvent pendingEvent = new PendingEvent(id, type, getContextString(context), payload, now());
43          publishedEvents.add(pendingEvent);
44          publishingEvent.fire(new PublishingEvent(id));
45      }
46  
47      private String getContextString(Properties context) {
48          if (context.isEmpty()) {
49              return null;
50          }
51          try (StringWriter writer = new StringWriter()) {
52              context.store(writer, null);
53              return writer.getBuffer().substring(writer.getBuffer().indexOf("\n"));
54          } catch (IOException e) {
55              LOGGER.warn("unexpected IOException while writing context", e);
56              return null;
57          }
58      }
59  
60  }

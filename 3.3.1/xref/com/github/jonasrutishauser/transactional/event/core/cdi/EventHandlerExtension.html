64      public Annotation getHandlerQualifier(EventTypeResolver typeResolver, String type) {
65          for (TypedEventHandler handlerQualifier : handlerQualifiers) {
66              if (type.equals(typeResolver.resolve(handlerQualifier.value()))) {
67                  return handlerQualifier;
68              }
69          }
70          return EventHandlerLiteral.of(type);

73      void useBetterLifecycleEventsIfPossible(
74              @Observes @Priority(LIBRARY_BEFORE) ProcessObserverMethod<Object, ? extends Startup> event) {
75          try {
76              if (event.getObserverMethod().getObservedQualifiers().stream().filter(Initialized.class::isInstance)
77                      .map(Initialized.class::cast).map(Initialized::value).anyMatch(isEqual(ApplicationScoped.class))) {
78                  event.configureObserverMethod() //
79                          .observedType(jakarta.enterprise.event.Startup.class) //
80                          .qualifiers();
81              }
82              if (event.getObserverMethod().getObservedQualifiers().stream().filter(BeforeDestroyed.class::isInstance)
83                      .map(BeforeDestroyed.class::cast).map(BeforeDestroyed::value)
84                      .anyMatch(isEqual(ApplicationScoped.class))) {
85                  event.configureObserverMethod().observedType(Shutdown.class) //
86                          .qualifiers();
87              }
88          } catch (NoClassDefFoundError e) {
89              // ignore
90          }
91      }

93      <T extends AbstractHandler<?>> void processTypedHandlers(
94              @Observes @Priority(LIBRARY_AFTER) @WithAnnotations(EventHandler.class) ProcessAnnotatedType<T> typeEvent) {
95          AnnotatedType<T> type = typeEvent.getAnnotatedType();
96          if (type.isAnnotationPresent(EventHandler.class)
97                  && EventHandler.ABSTRACT_HANDLER_TYPE.equals(type.getAnnotation(EventHandler.class).eventType())) {
98              Class<?> eventType = (Class<?>) getAbstractHandlerType(type.getTypeClosure()).orElseThrow()
99                      .getActualTypeArguments()[0];
100             typeEvent.configureAnnotatedType().add(TypedEventHandler.Literal.of(eventType));
101         }
102     }

104     <T extends Handler> void processHandlers(@Observes @Priority(LIBRARY_AFTER) ProcessBean<T> beanEvent) {
105         Optional<EventHandler> annotation = beanEvent.getBean().getQualifiers().stream() //
106             .filter(a -> EventHandler.class.equals(a.annotationType())) //
107             .map(EventHandler.class::cast) //
108             .findAny();
109         if (annotation.isEmpty()) {
110             beanEvent.addDefinitionError(
111                     new IllegalStateException("EventHandler annotation is missing on bean " + beanEvent.getBean()));
112         } else if (!beanEvent.getBean().getTypes().contains(Handler.class)) {
113             beanEvent.addDefinitionError(
114                     new IllegalStateException(Handler.class + " type is missing on bean " + beanEvent.getBean()));
115         } else {
116             Optional<TypedEventHandler> typedEventHandler = beanEvent.getBean().getQualifiers().stream() //
117                     .filter(a -> TypedEventHandler.class.equals(a.annotationType())) //
118                     .map(TypedEventHandler.class::cast) //
119                     .findAny();
120             if (EventHandler.ABSTRACT_HANDLER_TYPE.equals(annotation.get().eventType())) {
121                 if (typedEventHandler.isEmpty()) {
122                     beanEvent.addDefinitionError(new IllegalStateException("AbstractHandler type is missing on bean "
123                             + beanEvent.getBean() + " with implicit event type"));
124                 } else {
125                     handlerQualifiers.add(typedEventHandler.get());
126                 }
127             }
128 
129         }
130     }

132     <T> void processHandlerMethods(@Observes @Priority(LIBRARY_AFTER) ProcessManagedBean<T> beanEvent) {
133         try {
134             beanEvent.getAnnotatedBeanClass().getMethods().stream() //
135                     .filter(m -> m.isAnnotationPresent(EventHandler.class)) //
136                     .forEach(m -> addHandlerMethod(beanEvent, m));
137         } catch (NoSuchMethodError e) {
138             beanEvent.addDefinitionError(
139                     new IllegalStateException("direct handler methods need a CDI version >= 4.1", e));
140         }
141     }

143     private <T> void addHandlerMethod(ProcessManagedBean<T> beanEvent, AnnotatedMethod<? super T> method) {
144         if (method.getParameters().size() != 1) {
145             beanEvent.addDefinitionError(
146                     new IllegalStateException("EventHandler method " + method + " must have exactly one argument"));
147         } else {
148             Invoker<T, ?> invoker = beanEvent.createInvoker(method).withInstanceLookup().build();
149             eventHandlerMethods.add(new EventHandlerMethod<>(invoker,
150                     method.getParameters().get(0).getJavaParameter().getType(), method.getAnnotation(EventHandler.class)));
151         }
152     }

159     void processEventDeserializerInjections(InjectionPoint injectionPoint) {
160         Type type = injectionPoint.getType();
161         if (type instanceof ParameterizedType
162                 && EventDeserializer.class.equals(((ParameterizedType) type).getRawType())) {
163             requiredEventDeserializers.add((ParameterizedType) type);

167     void addSyntheticEventHandlers(@Observes @Priority(LIBRARY_BEFORE) AfterBeanDiscovery event,
168             BeanManager beanManager) {
169         for (EventHandlerMethod<?> eventHandlerMethod : eventHandlerMethods) {
170             InjectionPoint injectionPoint = eventHandlerMethod.getEventDeserializerInjectionPoint(beanManager);
171             processEventDeserializerInjections(injectionPoint);
172             Annotation[] qualifiers = eventHandlerMethod.getEventHandlerQualifiers();
173             event.addBean().beanClass(SyntheticHandler.class) //
174                     .types(Handler.class) //
175                     .qualifiers(qualifiers) //
176                     .addInjectionPoint(injectionPoint) //
177                     .createWith(ctx -> eventHandlerMethod.createHandler(ctx, beanManager));
178         }
179     }

181     void addMissingEventDeserializers(@Observes @Priority(LIBRARY_AFTER) AfterBeanDiscovery event,
182             BeanManager beanManager) {
183         for (ParameterizedType type : requiredEventDeserializers) {
184             try {
185                 if (beanManager.resolve(beanManager.getBeans(type)) == null) {
186                     Type eventType = type.getActualTypeArguments()[0];
187                     Class<?> eventClass = eventType instanceof Class ? (Class<?>) eventType
188                             : (Class<?>) ((ParameterizedType) eventType).getRawType();
189                     genericSerializationEventTypes.add(eventClass);
190                     event.addBean() //
191                             .addType(type) //
192                             .addType(ExtendedEventDeserializer.class) //
193                             .scope(ApplicationScoped.class) //
194                             .qualifiers(Default.Literal.INSTANCE) //
195                             .produceWith(instance -> createDefaultEventDeserializer(
196                                     instance.select(GenericSerialization.class), eventClass));
197                 }
198             } catch (AmbiguousResolutionException e) {
199                 // ignore, will be reported later
200             }
201         }
202     }

204     void verifyGenericSerializationEventTypes(@Observes AfterDeploymentValidation event, BeanManager beanManager) {
205         List<GenericSerialization> serializations = new ArrayList<>();
206         Instance<GenericSerialization> instance = beanManager.createInstance().select(GenericSerialization.class,
207                 Literal.INSTANCE);
208         instance.forEach(serializations::add);
209         for (Class<?> eventType : genericSerializationEventTypes) {
210             if (serializations.stream().noneMatch(s -> s.accepts(eventType))) {
211                 event.addDeploymentProblem(
212                         new UnsatisfiedResolutionException("No GenericSerialization found for " + eventType));
213             }
214         }
215         serializations.forEach(instance::destroy);
216     }

218     public static <T> DefaultEventDeserializer<T> createDefaultEventDeserializer(Instance<GenericSerialization> instance,
219             Class<T> type) {
220         List<GenericSerialization> serializations = new ArrayList<>();
221         instance.forEach(serializations::add);
222         sort(serializations, comparing(GenericSerialization::priority));
223         DefaultEventDeserializer<T> result = null;
224         for (GenericSerialization serialization : serializations) {
225             if (result == null && serialization.accepts(type)) {
226                 result = new DefaultEventDeserializer<>(type, serialization);
227             } else {
228                 instance.destroy(serialization);
229             }
230         }
231         if (result == null) {
232             throw new UnsatisfiedResolutionException("No GenericSerialization found for " + type);
233         }
234         return result;
235     }

237     private Optional<ParameterizedType> getAbstractHandlerType(Set<Type> types) {
238         return types.stream() //
239                 .filter(ParameterizedType.class::isInstance) //
240                 .map(ParameterizedType.class::cast) //
241                 .filter(type -> AbstractHandler.class.equals(type.getRawType())) //
242                 .findAny();
243     }

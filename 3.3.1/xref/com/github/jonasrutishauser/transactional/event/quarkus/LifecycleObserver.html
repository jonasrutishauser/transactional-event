4   import io.quarkus.arc.ArcContainer;
5   import io.quarkus.arc.InstanceHandle;
6   import jakarta.enterprise.inject.build.compatible.spi.Parameters;
7   import jakarta.enterprise.inject.build.compatible.spi.SyntheticObserver;
8   import jakarta.enterprise.inject.spi.EventContext;
9   import jakarta.enterprise.invoke.Invoker;
10  
11  public class LifecycleObserver implements SyntheticObserver<Object> {
12  
13      public static final String INVOKER = "invoker";
14      public static final String TYPE = "type";
15  
16      @Override
17      @SuppressWarnings("unchecked")
18      public void observe(EventContext<Object> event, Parameters params) throws Exception {
19          Object instance = null;
20          Class<?> type = params.get(TYPE, Class.class);
21          if (type != null) {
22              ArcContainer container = Arc.container();
23              InstanceHandle<?> handle = container.instance(type);
24              instance = container.getActiveContext(handle.getBean().getScope()).get(handle.getBean());
25              if (instance == null) {
26                  return;
27              }
28          }
29          params.get(INVOKER, Invoker.class).invoke(instance, new Object[] {event.getEvent()});
30      }

32  }

78      private static class LifecycleObserverInfo {
79          private final MethodInfo method;
80          private final boolean instanceLookup;
81          private InvokerInfo invoker;
82  
83          LifecycleObserverInfo(MethodInfo method, boolean instanceLookup) {
84              this.method = method;
85              this.instanceLookup = instanceLookup;
86          }
87  
88          MethodInfo method() {
89              return method;
90          }
91  
92          InvokerInfo invoker() {
93              return invoker;
94          }
95  
96          int priority() {
97              return method.parameters().get(0).annotation(Priority.class).value().asInt();
98          }
99  
100         void createInvoker(BeanInfo bean, InvokerFactory invokerFactory) {
101             InvokerBuilder<InvokerInfo> builder = invokerFactory.createInvoker(bean, method);
102             if (instanceLookup) {
103                 builder = builder.withInstanceLookup();
104             }
105             invoker = builder.build();
106         }
107     }

126     private void handleApplicationScopeObservers(ClassConfig type, Types types) {
127         Type applicationScoped = types.of(ApplicationScoped.class);
128         type.methods().forEach(method -> {
129             if (!method.parameters().isEmpty()) {
130                 ParameterConfig firstParameter = method.parameters().get(0);
131                 if (firstParameter.info().hasAnnotation(Observes.class) && firstParameter.info()
132                         .hasAnnotation(annotation -> (Initialized.class.getName().equals(annotation.name())
133                                 || BeforeDestroyed.class.getName().equals(annotation.name()))
134                                 && applicationScoped.equals(annotation.value().asType()))) {
135                     firstParameter.removeAllAnnotations();
136                     if (firstParameter.info()
137                             .hasAnnotation(annotation -> Initialized.class.getName().equals(annotation.name()))) {
138                         startupObservers.add(new LifecycleObserverInfo(method.info(), true));
139                     } else {
140                         shutdownObservers.add(new LifecycleObserverInfo(method.info(), false));
141                     }
142                 }
143             }
144         });
145     }

147     @Registration(types = Startup.class)
148     @Priority(LIBRARY_AFTER)
149     public void processStartupBeans(BeanInfo beanInfo, InvokerFactory invokerFactory) {
150         Collection<MethodInfo> methods = beanInfo.declaringClass().methods();
151         concat(startupObservers.stream(), shutdownObservers.stream()) //
152                 .filter(observer -> methods.contains(observer.method())) //
153                 .forEach(observer -> observer.createInvoker(beanInfo, invokerFactory));
154     }

156     @Synthesis
157     @Priority(LIBRARY_BEFORE)
158     public void addSyntheticLifecycleObservers(SyntheticComponents components, Types types) {
159         for (LifecycleObserverInfo observer : startupObservers) {
160             if (observer.invoker() != null) {
161                 components.addObserver(types.of(jakarta.enterprise.event.Startup.class)) //
162                         .priority(observer.priority()) //
163                         .declaringClass(observer.method().declaringClass()) //
164                         .observeWith(LifecycleObserver.class) //
165                         .withParam(LifecycleObserver.INVOKER, observer.invoker());
166             }
167         }
168         for (LifecycleObserverInfo observer : shutdownObservers) {
169             if (observer.invoker() != null) {
170                 components.addObserver(types.of(Shutdown.class)) //
171                         .priority(observer.priority()) //
172                         .declaringClass(observer.method().declaringClass()) //
173                         .observeWith(LifecycleObserver.class) //
174                         .withParam(LifecycleObserver.INVOKER, observer.invoker()) //
175                         .withParam(LifecycleObserver.TYPE, observer.method().declaringClass());
176             }
177         }

180     @Enhancement(types = AbstractHandler.class, withSubtypes = true, withAnnotations = EventHandler.class)
181     @Priority(LIBRARY_AFTER)
182     public void processTypedHandlers(ClassConfig typeConfig, Types types, Messages messages) {
183         ClassInfo type = typeConfig.info();
184         if (type.hasAnnotation(EventHandler.class) && EventHandler.ABSTRACT_HANDLER_TYPE
185                 .equals(type.annotation(EventHandler.class).member(EVENT_TYPE).asString())) {
186             Optional<ClassType> eventType = type.methods().stream() //
187                     .filter(m -> "handle".equals(m.name())) //
188                     .filter(m -> types.ofVoid().equals(m.returnType())) //
189                     .filter(m -> !m.isAbstract()) //
190                     .filter(m -> !m.isStatic()) //
191                     .filter(m -> m.parameters().size() == 1) //
192                     .map(MethodInfo::parameters) //
193                     .map(params -> params.get(0)) //
194                     .map(ParameterInfo::type) //
195                     .filter(Type::isClass) //
196                     .map(Type::asClass) //
197                     .findAny();
198             if (eventType.isEmpty()) {
199                 messages.warn("Failed to determine event type", type);
200             } else {
201                 typeConfig.addAnnotation(AnnotationBuilder.of(TypedEventHandler.class).value(eventType.get()).build());
202             }
203         }
204     }

206     @Registration(types = Handler.class)
207     @Priority(LIBRARY_AFTER)
208     public void processHandlers(BeanInfo beanInfo, Messages messages) {
209         Optional<AnnotationInfo> eventHandlerAnnotation = beanInfo.qualifiers().stream()
210                 .filter(annotation -> EventHandler.class.getName().equals(annotation.name())).findAny();
211         if (eventHandlerAnnotation.isEmpty()) {
212             messages.error("EventHandler annotation is missing on bean", beanInfo);
213         } else {
214             if (EventHandler.ABSTRACT_HANDLER_TYPE
215                     .equals(eventHandlerAnnotation.get().member(EVENT_TYPE).asString())) {
216                 Optional<AnnotationInfo> typedEventHandler = beanInfo.qualifiers().stream() //
217                         .filter(a -> TypedEventHandler.class.getName().equals(a.name())) //
218                         .findAny();
219                 if (typedEventHandler.isEmpty()) {
220                     messages.error("AbstractHandler type is missing on bean with implicit event type", beanInfo);
221                 } else {
222                     handledTypes.add(typedEventHandler.get().value().asType().asClass().declaration());
223                 }
224             }
225         }
226     }

238     private void addHandlerMethod(BeanInfo beanInfo, MethodInfo method, InvokerFactory invokerFactory, Messages messages) {
239         if (method.parameters().size() != 1) {
240             messages.error("EventHandler method must have exactly one argument", method);
241         } else {
242             InvokerInfo invoker = invokerFactory.createInvoker(beanInfo, method).withInstanceLookup().build();
243             eventHandlerMethods.put(method, invoker);
244         }
245     }

247     @Synthesis
248     @Priority(LIBRARY_BEFORE)
249     public void addSyntheticEventHandlers(SyntheticComponents components, Types types) {
250         for (Entry<MethodInfo, InvokerInfo> eventHandlerMethod : eventHandlerMethods.entrySet()) {
251             ClassType eventType = eventHandlerMethod.getKey().parameters().get(0).type().asClass();
252             components.addBean(Handler.class) //
253                     .type(Handler.class) //
254                     .qualifier(eventHandlerMethod.getKey().annotation(EventHandler.class)) //
255                     .qualifier(AnnotationBuilder.of(TypedEventHandler.class).value(eventType.declaration()).build()) //
256                     .createWith(SyntheticHandlerCreator.class) //
257                     .withParam("type", eventType.declaration()) //
258                     .withParam("invoker", eventHandlerMethod.getValue());
259             if (EventHandler.ABSTRACT_HANDLER_TYPE.equals(
260                     eventHandlerMethod.getKey().annotation(EventHandler.class).member(EVENT_TYPE).asString())) {
261                 handledTypes.add(eventType.declaration());
262             }
263             requiredEventDeserializers.add(types.parameterized(EventDeserializer.class, eventType));
264         }
265     }

281     @SuppressWarnings("unchecked")
282     private <T> void addCreator(SyntheticBeanBuilder<T> builder, Class<?> creatorHost) {
283         builder.createWith(
284                 Arrays.stream(creatorHost.getNestMembers()).filter(SyntheticBeanCreator.class::isAssignableFrom)
285                         .map(c -> (Class<? extends SyntheticBeanCreator<T>>) c).findAny()
286                         .orElseThrow(IllegalStateException::new));
287     }

307     @Synthesis
308     @Priority(LIBRARY_AFTER)
309     public void addMissingEventDeserializers(SyntheticComponents components) {
310         requiredEventDeserializers.removeAll(declaredEventDeserializers);
311         for (ParameterizedType type : requiredEventDeserializers) {
312             Type eventType = type.typeArguments().get(0);
313             ClassType eventClass = eventType.isClass() ? eventType.asClass()
314                     : eventType.asParameterizedType().genericClass();
315             components.addBean(DefaultEventDeserializer.class) //
316                 .type(type) //
317                 .type(ExtendedEventDeserializer.class) //
318                 .scope(Singleton.class) //
319                 .qualifier(Default.Literal.INSTANCE) //
320                 .createWith(DefaultEventDeserializerCreator.class) //
321                 .withParam(DefaultEventDeserializerCreator.TYPE, eventClass.declaration());
322         }

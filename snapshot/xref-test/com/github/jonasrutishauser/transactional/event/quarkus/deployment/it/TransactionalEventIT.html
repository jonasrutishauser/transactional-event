43      @Test
44      void testFailure() {
45          publisher.publish("test failure");
46          for (int i = 10; i < 50; i++) {
47              publisher.publish("failure " + i);
48          }
49  
50          await().atMost(1, MINUTES).until(processedMessagesContains("test failure"));
51          for (int i = 10; i < 50; i++) {
52              await().until(processedMessagesContains("failure " + i));
53          }
54      }
55  
56      private Callable<Boolean> processedMessagesContains(String content) {
57          return () -> publisher.getMessages().contains(content);
58      }

1   package com.github.jonasrutishauser.transactional.event.core.store;
2   
3   import jakarta.enterprise.context.Dependent;
4   import jakarta.enterprise.context.control.ActivateRequestContext;
5   import jakarta.enterprise.event.Event;
6   import jakarta.inject.Inject;
7   
8   import org.apache.logging.log4j.LogManager;
9   import org.apache.logging.log4j.Logger;
10  import org.eclipse.microprofile.metrics.annotation.ConcurrentGauge;
11  
12  import com.github.jonasrutishauser.transactional.event.api.monitoring.ProcessingFailedEvent;
13  import com.github.jonasrutishauser.transactional.event.api.monitoring.ProcessingSuccessEvent;
14  
15  @Dependent
16  class WorkerImpl implements Worker {
17  
18      private static final Logger LOGGER = LogManager.getLogger();
19  
20      private final TransactionalWorker transactional;
21  
22      private final Event<ProcessingSuccessEvent> processingSuccessEvent;
23      private final Event<ProcessingFailedEvent> processingFailedEvent;
24  
25      @Inject
26      WorkerImpl(TransactionalWorker transactional, Event<ProcessingSuccessEvent> processingSuccessEvent,
27              Event<ProcessingFailedEvent> processingFailedEvent) {
28          this.transactional = transactional;
29          this.processingSuccessEvent = processingSuccessEvent;
30          this.processingFailedEvent = processingFailedEvent;
31      }
32  
33      @ActivateRequestContext
34      @ConcurrentGauge(name = "com.github.jonasrutishauser.transaction.event.processing",
35              description = "number of events in processing", absolute = true)
36      public boolean process(String eventId) {
37          try {
38              processAttempt(eventId);
39              transactional.process(eventId);
40              processSuccess(eventId);
41              return true;
42          } catch (Exception e) {
43              processAttemptFailed(eventId, e);
44              transactional.processFailed(eventId);
45              return false;
46          }
47      }
48  
49      private void processAttemptFailed(String eventId, Exception e) {
50          processingFailedEvent.fire(new ProcessingFailedEvent(eventId, e));
51          LOGGER.warn("Failed to process event with id '{}'", eventId, e);
52      }
53  
54      protected void processSuccess(String eventId) {
55          processingSuccessEvent.fire(new ProcessingSuccessEvent(eventId));
56          LOGGER.debug("sucessfully processed event with id '{}'", eventId);
57      }
58  
59      protected void processAttempt(String eventId) {
60          LOGGER.debug("processing event with id '{}'", eventId);
61      }
62  
63  }

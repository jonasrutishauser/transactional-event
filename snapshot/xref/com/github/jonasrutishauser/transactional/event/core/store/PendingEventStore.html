71      @PostConstruct
72      void initSqlQueries() {
73          QueryAdapter adapter = queryAdapterFactory.getQueryAdapter();
74          insertSQL = "INSERT INTO " + configuration.getTableName()
75                  + " (id, event_type, payload, published_at, tries, lock_owner, locked_until) VALUES (?, ?, ?, ?, ?, ?, ?)";
76          readSQL = "SELECT * FROM " + configuration.getTableName() + " WHERE id=? FOR UPDATE";
77          deleteSQL = "DELETE FROM " + configuration.getTableName() + " WHERE id=? AND lock_owner=?";
78          updateSQL = "UPDATE " + configuration.getTableName() + " SET tries=?, lock_owner=?, locked_until=? WHERE id=?";
79          updateSQLwithLockOwner = updateSQL + " AND lock_owner=?";
80          aquireSQL = adapter.fixLimits(adapter.addSkipLocked("SELECT id, tries FROM " + configuration.getTableName()
81                  + " WHERE locked_until<=? {LIMIT ?} FOR UPDATE"));
82          String readBlocked = "SELECT * FROM " + configuration.getTableName() + " WHERE locked_until=" + Long.MAX_VALUE;
83          readBlockedSQL = adapter.fixLimits(readBlocked + " {LIMIT ?}");
84          readBlockedForUpdateSQL = adapter.addSkipLocked(readBlocked + " AND id=? FOR UPDATE");
85      }

87      @Override
88      @Transactional
89      public boolean unblock(String eventId) {
90          boolean result;
91          try (Connection connection = dataSource.getConnection();
92               PreparedStatement readStatement = connection.prepareStatement(readBlockedForUpdateSQL);
93               ResultSet resultSet = executeQuery(readStatement, eventId);
94               PreparedStatement updateStatement = connection.prepareStatement(updateSQL)) {
95              if (resultSet.next()) {
96                  updateStatement.setInt(1, 0);
97                  updateStatement.setNull(2, VARCHAR);
98                  updateStatement.setLong(3, lockOwner.getUntilForRetry(0, eventId));
99                  updateStatement.setString(4, eventId);
100                 result = updateStatement.executeUpdate() > 0;
101             } else {
102                 result = false;
103             }
104         } catch (SQLException exception) {
105             LOGGER.error("failed to unblock event '{}'", eventId, exception);
106             result = false;
107         }
108         return result;
109     }

111     @Override
112     @Transactional
113     public Collection<BlockedEvent> getBlockedEvents(int maxElements) {
114         Collection<BlockedEvent> result = new ArrayList<>();
115         try (Connection connection = dataSource.getConnection();
116              PreparedStatement statement = connection.prepareStatement(readBlockedSQL);
117              ResultSet resultSet = executeQuery(statement, maxElements)) {
118             while (resultSet.next()) {
119                 result.add(new BlockedEvent(resultSet.getString("id"), //
120                         resultSet.getString("event_type"), //
121                         resultSet.getString("payload"), //
122                         resultSet.getTimestamp("published_at").toLocalDateTime()));
123             }
124         } catch (SQLException exception) {
125             LOGGER.error("failed to read blocked events", exception);
126         }
127         return result;
128     }

130     @Transactional(MANDATORY)
131     void store(@Observes(during = BEFORE_COMPLETION) EventsPublished events) {
132         String errorMessage = "failed to insert pending events";
133         try (Connection connection = dataSource.getConnection();
134              PreparedStatement statement = connection.prepareStatement(insertSQL)) {
135             for (PendingEvent event : events.getEvents()) {
136                 statement.setString(1, event.getId());
137                 statement.setString(2, event.getType());
138                 statement.setString(3, event.getPayload());
139                 statement.setTimestamp(4, Timestamp.valueOf(event.getPublishedAt()));
140                 statement.setInt(5, event.getTries());
141                 statement.setString(6, lockOwner.getId());
142                 statement.setLong(7, lockOwner.getUntilToProcess());
143                 statement.addBatch();
144             }
145             int[] result = statement.executeBatch();
146             if (result.length != events.getEvents().size()
147                     || Arrays.stream(result).anyMatch(count -> count != 1 && count != SUCCESS_NO_INFO)) {
148                 LOGGER.error("failed to insert pending events (results: {})", result);
149                 throw new IllegalStateException(errorMessage);
150             }
151         } catch (SQLException exception) {
152             LOGGER.error(errorMessage, exception);
153             throw new IllegalStateException(errorMessage, exception);
154         }
155     }

157     @Transactional(value = MANDATORY, dontRollbackOn = NoSuchElementException.class)
158     public PendingEvent getAndLockEvent(String id) {
159         String errorMessage = "failed to read pending event with id " + id;
160         try (Connection connection = dataSource.getConnection();
161              PreparedStatement statement = connection.prepareStatement(readSQL);
162              ResultSet resultSet = executeQueryWithTimeout(statement, id, 1)) {
163             if (!resultSet.next()) {
164                 throw new NoSuchElementException(errorMessage);
165             }
166             String owner = resultSet.getString("lock_owner");
167             long lockedUntil = resultSet.getLong("locked_until");
168             if (!lockOwner.isOwningForProcessing(owner, lockedUntil)) {
169                 throw new ConcurrentModificationException("No longer the owner");
170             }
171             return new PendingEvent(id, resultSet.getString("event_type"), resultSet.getString("payload"),
172                     resultSet.getTimestamp("published_at").toLocalDateTime(), resultSet.getInt("tries"));
173         } catch (SQLException exception) {
174             LOGGER.error(errorMessage, exception);
175             throw new IllegalStateException(errorMessage);
176         }
177     }

179     @Transactional(MANDATORY)
180     public void delete(PendingEvent event) {
181         try (Connection connection = dataSource.getConnection();
182              PreparedStatement statement = connection.prepareStatement(deleteSQL)) {
183             statement.setString(1, event.getId());
184             statement.setString(2, lockOwner.getId());
185             statement.setQueryTimeout(10);
186             if (statement.executeUpdate() < 1) {
187                 throw new NoSuchElementException("failed to delete pending event with id " + event.getId());
188             }
189         } catch (SQLException exception) {
190             String errorMessage = "failed to delete pending event with id " + event.getId();
191             LOGGER.error(errorMessage, exception);
192             throw new IllegalStateException(errorMessage);
193         }
194     }

196     @Transactional(MANDATORY)
197     public void updateForRetry(PendingEvent event) {
198         try (Connection connection = dataSource.getConnection();
199              PreparedStatement statement = connection.prepareStatement(updateSQLwithLockOwner)) {
200             statement.setInt(1, event.getTries() + 1);
201             statement.setNull(2, VARCHAR);
202             statement.setLong(3, lockOwner.getUntilForRetry(event.getTries(), event.getId()));
203             statement.setString(4, event.getId());
204             statement.setString(5, lockOwner.getId());
205             statement.setQueryTimeout(10);
206             if (statement.executeUpdate() < 1) {
207                 throw new NoSuchElementException("failed to update pending event with id " + event.getId());
208             }
209         } catch (SQLException exception) {
210             String errorMessage = "failed to update pending event with id " + event.getId();
211             LOGGER.error(errorMessage, exception);
212             throw new IllegalStateException(errorMessage);
213         }
214     }

216     @Transactional
217     public Set<String> aquire() {
218         Set<String> result = new HashSet<>();
219         try (Connection connection = dataSource.getConnection();
220              PreparedStatement aquireStatement = connection.prepareStatement(aquireSQL);
221              ResultSet resultSet = executeQuery(aquireStatement, lockOwner.getMinUntilForAquire(),
222                      configuration.getMaxAquire());
223              PreparedStatement updateStatement = connection.prepareStatement(updateSQL)) {
224             while (resultSet.next()) {
225                 result.add(resultSet.getString("id"));
226                 updateStatement.setInt(1, resultSet.getInt("tries"));
227                 updateStatement.setString(2, lockOwner.getId());
228                 updateStatement.setLong(3, lockOwner.getUntilToProcess());
229                 updateStatement.setString(4, resultSet.getString("id"));
230                 updateStatement.addBatch();
231             }
232             if (!result.isEmpty()) {
233                 int[] res = updateStatement.executeBatch();
234                 if (res.length != result.size()
235                         || Arrays.stream(res).anyMatch(count -> count != 1 && count != SUCCESS_NO_INFO)) {
236                     LOGGER.warn("failed to aquire pending events (update failed; results: {})", res);
237                     result = emptySet();
238                 }
239             }
240         } catch (SQLException exception) {
241             LOGGER.warn("failed to aquire pending events", exception);
242             result = emptySet();
243         }
244         return result;
245     }

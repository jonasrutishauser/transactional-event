78      @Enhancement(types = com.github.jonasrutishauser.transactional.event.core.cdi.Startup.class, withSubtypes = true)
79      @Priority(LIBRARY_AFTER)
80      public void fixStaticInitStartup(ClassConfig type) {
81          if (!com.github.jonasrutishauser.transactional.event.core.cdi.Startup.class.getName().equals(type.info().name())) {
82              changeInitializedApplicationScopedObserverToStartup(type);
83          }
84      }

86      private void changeInitializedApplicationScopedObserverToStartup(ClassConfig type) {
87          type.addAnnotation(Startup.class);
88          type.methods().forEach(method -> {
89              if (!method.parameters().isEmpty()) {
90                  ParameterConfig firstParameter = method.parameters().get(0);
91                  if (firstParameter.info().hasAnnotation(Observes.class) && firstParameter.info()
92                          .hasAnnotation(annotation -> Initialized.class.getName().equals(annotation.name())
93                                  && annotation.value().asType().isClass() && ApplicationScoped.class.getName()
94                                          .equals(annotation.value().asType().asClass().declaration().name()))) {
95                      firstParameter.removeAllAnnotations();
96                  }
97              }
98          });
99      }

101     @Enhancement(types = AbstractHandler.class, withSubtypes = true, withAnnotations = EventHandler.class)
102     @Priority(LIBRARY_AFTER)
103     public void processTypedHandlers(ClassConfig typeConfig, Types types, Messages messages) {
104         ClassInfo type = typeConfig.info();
105         if (type.hasAnnotation(EventHandler.class) && EventHandler.ABSTRACT_HANDLER_TYPE
106                 .equals(type.annotation(EventHandler.class).member(EVENT_TYPE).asString())) {
107             Optional<ClassType> eventType = type.methods().stream() //
108                     .filter(m -> "handle".equals(m.name())) //
109                     .filter(m -> types.ofVoid().equals(m.returnType())) //
110                     .filter(m -> !m.isAbstract()) //
111                     .filter(m -> !m.isStatic()) //
112                     .filter(m -> m.parameters().size() == 1) //
113                     .map(MethodInfo::parameters) //
114                     .map(params -> params.get(0)) //
115                     .map(ParameterInfo::type) //
116                     .filter(Type::isClass) //
117                     .map(Type::asClass) //
118                     .findAny();
119             if (eventType.isEmpty()) {
120                 messages.warn("Failed to determine event type", type);
121             } else {
122                 typeConfig.addAnnotation(AnnotationBuilder.of(TypedEventHandler.class).value(eventType.get()).build());
123             }
124         }
125     }

127     @Registration(types = Handler.class)
128     @Priority(LIBRARY_AFTER)
129     public void processHandlers(BeanInfo beanInfo, Messages messages) {
130         Optional<AnnotationInfo> eventHandlerAnnotation = beanInfo.qualifiers().stream()
131                 .filter(annotation -> EventHandler.class.getName().equals(annotation.name())).findAny();
132         if (eventHandlerAnnotation.isEmpty()) {
133             messages.error("EventHandler annotation is missing on bean", beanInfo);
134         } else {
135             if (EventHandler.ABSTRACT_HANDLER_TYPE
136                     .equals(eventHandlerAnnotation.get().member(EVENT_TYPE).asString())) {
137                 Optional<AnnotationInfo> typedEventHandler = beanInfo.qualifiers().stream() //
138                         .filter(a -> TypedEventHandler.class.getName().equals(a.name())) //
139                         .findAny();
140                 if (typedEventHandler.isEmpty()) {
141                     messages.error("AbstractHandler type is missing on bean with implicit event type", beanInfo);
142                 } else {
143                     handledTypes.add(typedEventHandler.get().value().asType().asClass().declaration());
144                 }
145             }
146         }
147     }

149     @Registration(types = Object.class)

152         if (beanInfo.isClassBean()) {
153             beanInfo.declaringClass().methods().stream() //
154                 .filter(m -> m.hasAnnotation(EventHandler.class)) //
155                 .forEach(m -> addHandlerMethod(beanInfo, m, invokerFactory, messages));
156         }

159     private void addHandlerMethod(BeanInfo beanInfo, MethodInfo method, InvokerFactory invokerFactory, Messages messages) {
160         if (method.parameters().size() != 1) {
161             messages.error("EventHandler method must have exactly one argument", method);
162         } else {
163             InvokerInfo invoker = invokerFactory.createInvoker(beanInfo, method).withInstanceLookup().build();
164             eventHandlerMethods.put(method, invoker);
165         }
166     }

168     @Synthesis
169     @Priority(LIBRARY_BEFORE)
170     public void addSyntheticEventHandlers(SyntheticComponents components, Types types) {
171         for (Entry<MethodInfo, InvokerInfo> eventHandlerMethod : eventHandlerMethods.entrySet()) {
172             ClassType eventType = eventHandlerMethod.getKey().parameters().get(0).type().asClass();
173             components.addBean(Handler.class) //
174                     .type(Handler.class) //
175                     .qualifier(eventHandlerMethod.getKey().annotation(EventHandler.class)) //
176                     .qualifier(AnnotationBuilder.of(TypedEventHandler.class).value(eventType.declaration()).build()) //
177                     .createWith(SyntheticHandlerCreator.class) //
178                     .withParam("type", eventType.declaration()) //
179                     .withParam("invoker", eventHandlerMethod.getValue());
180             if (EventHandler.ABSTRACT_HANDLER_TYPE.equals(
181                     eventHandlerMethod.getKey().annotation(EventHandler.class).member(EVENT_TYPE).asString())) {
182                 handledTypes.add(eventType.declaration());
183             }
184             requiredEventDeserializers.add(types.parameterized(EventDeserializer.class, eventType));
185         }
186     }

202     @SuppressWarnings("unchecked")
203     private <T> void addCreator(SyntheticBeanBuilder<T> builder, Class<?> creatorHost) {
204         builder.createWith(
205                 Arrays.stream(creatorHost.getNestMembers()).filter(SyntheticBeanCreator.class::isAssignableFrom)
206                         .map(c -> (Class<? extends SyntheticBeanCreator<T>>) c).findAny()
207                         .orElseThrow(IllegalStateException::new));
208     }

213         for (InjectionPointInfo injectionPoint : beanInfo.injectionPoints()) {

228     @Synthesis
229     @Priority(LIBRARY_AFTER)
230     public void addMissingEventDeserializers(SyntheticComponents components) {
231         requiredEventDeserializers.removeAll(declaredEventDeserializers);
232         for (ParameterizedType type : requiredEventDeserializers) {
233             Type eventType = type.typeArguments().get(0);
234             ClassType eventClass = eventType.isClass() ? eventType.asClass()
235                     : eventType.asParameterizedType().genericClass();
236             components.addBean(DefaultEventDeserializer.class) //
237                 .type(type) //
238                 .type(ExtendedEventDeserializer.class) //
239                 .scope(Singleton.class) //
240                 .qualifier(Default.Literal.INSTANCE) //
241                 .createWith(DefaultEventDeserializerCreator.class) //
242                 .withParam(DefaultEventDeserializerCreator.TYPE, eventClass.declaration());
243         }

6   import jakarta.enterprise.context.ApplicationScoped;
7   import jakarta.enterprise.inject.Any;
8   import jakarta.enterprise.inject.Instance;
9   import jakarta.inject.Inject;

11  import com.github.jonasrutishauser.transactional.event.api.serialization.EventSerializer;
12  
13  @ApplicationScoped
14  class Serializer {
15  
16      private final Map<Class<?>, EventSerializer<?>> serializers = new ConcurrentHashMap<>();
17  
18      Serializer() {
19          // proxy only
20      }
21  
22      @Inject
23      Serializer(@Any Instance<EventSerializer<?>> serializers) {
24          for (EventSerializer<?> serializer : serializers) {
25              this.serializers.put(serializer.getType(), serializer);
26          }
27      }
28  
29      String serialize(Object event) {
30          return serialize(getSerializer(event), event);
31      }
32  
33      private <T> String serialize(EventSerializer<T> serializer, Object event) {
34          return serializer.serialize(serializer.getType().cast(event));
35      }
36  
37      private EventSerializer<?> getSerializer(Object event) {
38          return serializers.computeIfAbsent(event.getClass(), this::getSerializer);
39      }
40  
41      private EventSerializer<?> getSerializer(Class<?> eventType) {
42          for (Class<?> type = eventType.getSuperclass(); type != null; type = type.getSuperclass()) {
43              if (serializers.containsKey(type)) {
44                  return serializers.get(type);
45              }
46          }
47          throw new IllegalArgumentException("Serializer not found for " + eventType.getName());
48      }
49  
50  }

15  import jakarta.annotation.PostConstruct;
16  import jakarta.annotation.PreDestroy;
17  import jakarta.annotation.Priority;
18  import jakarta.enterprise.context.ApplicationScoped;
19  import jakarta.enterprise.context.BeforeDestroyed;
20  import jakarta.enterprise.context.Initialized;
21  import jakarta.enterprise.event.Observes;
22  import jakarta.inject.Inject;
23  
24  @ApplicationScoped
25  class Lifecycle implements Startup {
26  
27      private static final Logger LOGGER = LogManager.getLogger();
28  
29      private final Dispatcher dispatcher;
30      private final EventExecutor executor;
31  
32      private Task scheduled;
33  
34      Lifecycle() {
35          this(null, null);
36      }
37  
38      @Inject
39      Lifecycle(Dispatcher dispatcher, EventExecutor executor) {
40          this.dispatcher = dispatcher;
41          this.executor = executor;
42      }
43  
44      void directDispatch(@Observes(during = AFTER_SUCCESS) @Priority(LIBRARY_BEFORE) EventsPublished events) {
45          dispatcher.processDirect(events);
46      }
47  
48      void startup(@Observes @Priority(LIBRARY_AFTER + 500) @Initialized(ApplicationScoped.class) Object event) {
49          LOGGER.debug("initialized");
50      }
51  
52      @PostConstruct
53      void startup() {
54          scheduled = executor.schedule(this::safeSchedule, dispatcher::dispatchInterval);
55      }
56  
57      private void safeSchedule() {
58          try {
59              dispatcher.schedule();
60          } catch (RuntimeException e) {
61              LOGGER.warn("Failed to schedule event processing", e);
62          }
63      }
64  
65      void shutdown(@Observes(notifyObserver = IF_EXISTS) @Priority(LIBRARY_BEFORE) @BeforeDestroyed(ApplicationScoped.class) Object event) {
66          stop();
67      }
68  
69      @PreDestroy
70      void stop() {
71          if (scheduled != null) {
72              scheduled.cancel();
73              scheduled = null;
74          }
75      }
76  
77  }

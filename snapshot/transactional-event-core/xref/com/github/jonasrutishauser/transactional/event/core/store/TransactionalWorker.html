5   import javax.enterprise.context.Dependent;
6   import javax.enterprise.inject.Any;
7   import javax.enterprise.inject.Instance;
8   import javax.enterprise.util.AnnotationLiteral;
9   import javax.inject.Inject;
10  import javax.transaction.Transactional;
11  
12  import com.github.jonasrutishauser.transactional.event.api.EventTypeResolver;
13  import com.github.jonasrutishauser.transactional.event.api.handler.EventHandler;
14  import com.github.jonasrutishauser.transactional.event.api.handler.Handler;
15  import com.github.jonasrutishauser.transactional.event.core.PendingEvent;
16  import com.github.jonasrutishauser.transactional.event.core.cdi.EventHandlerExtension;
17  
18  @Dependent
19  class TransactionalWorker {
20  
21      private final PendingEventStore store;
22      private final Instance<Handler> handlers;
23      private final EventHandlerExtension handlerExtension;
24      private final EventTypeResolver typeResolver;
25  
26      TransactionalWorker() {
27          this(null, null, null, null);
28      }
29  
30      @Inject
31      TransactionalWorker(PendingEventStore store, @Any Instance<Handler> handlers, EventHandlerExtension handlerExtension,
32              EventTypeResolver typeResolver) {
33          this.store = store;
34          this.handlers = handlers;
35          this.handlerExtension = handlerExtension;
36          this.typeResolver = typeResolver;
37      }
38  
39      @Transactional
40      public void process(String eventId) {
41          PendingEvent event = store.getAndLockEvent(eventId);
42          Optional<Class<? extends Handler>> handlerClassWithImplicitType = handlerExtension
43                  .getHandlerClassWithImplicitType(typeResolver, event.getType());
44          Instance<? extends Handler> handlerInstance;
45          if (handlerClassWithImplicitType.isPresent()) {
46              handlerInstance = handlers.select(handlerClassWithImplicitType.get());
47          } else {
48              handlerInstance = handlers.select(new EventHandlerLiteral() {
49                  @Override
50                  public String eventType() {
51                      return event.getType();
52                  }
53              });
54          }
55          process(event, handlerInstance);
56          store.delete(event);
57      }
58  
59      @Transactional
60      public void processFailed(String eventId) {
61          PendingEvent event = store.getAndLockEvent(eventId);
62          store.updateForRetry(event);
63      }
64  
65      private <T extends Handler> void process(PendingEvent event, Instance<T> handlerInstance) {
66          T handler = handlerInstance.get();
67          try {
68              handler.handle(event.getPayload());
69          } finally {
70              handlerInstance.destroy(handler);
71          }
72      }
73  
74      @SuppressWarnings("serial")
75      private abstract static class EventHandlerLiteral extends AnnotationLiteral<EventHandler> implements EventHandler {}
76  
77  }

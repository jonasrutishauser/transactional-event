22  import io.opentelemetry.context.propagation.TextMapPropagator;
23  
24  @Decorator
25  @Priority(LIBRARY_AFTER)
26  public class InstrumentedPublisher implements ContextualPublisher {
27  
28      private final ContextualPublisher delegate;
29  
30      private final Tracer tracer;
31      private final TextMapPropagator propagator;
32  
33      @Inject
34      InstrumentedPublisher(@Delegate @Any ContextualPublisher delegate, @Events Tracer tracer,
35              @Events TextMapPropagator propagator) {
36          this.delegate = delegate;
37          this.tracer = tracer;
38          this.propagator = propagator;
39      }
40  
41      @Override
42      public void publish(String id, String type, Properties context, String payload) {
43          Span span = tracer.spanBuilder(type + " send") //
44                  .setSpanKind(PRODUCER) //
45                  .setAttribute("messaging.system", "transactional-event") //
46                  .setAttribute("messaging.destination", type) //
47                  .setAttribute("messaging.message_id", id) //
48                  .setAttribute("messaging.message_payload_size_bytes", payload.getBytes(UTF_8).length) //
49                  .startSpan();
50          try (Scope unused = span.makeCurrent()) {
51              propagator.inject(Context.current(), context, Properties::setProperty);
52              delegate.publish(id, type, context, payload);
53          } finally {
54              span.end();
55          }
56      }

58  }

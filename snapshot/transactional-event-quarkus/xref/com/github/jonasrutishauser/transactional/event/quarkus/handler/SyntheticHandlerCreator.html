17  public class SyntheticHandlerCreator implements SyntheticBeanCreator<Handler> {
18  
19      private static final TypeLiteral<EventDeserializer<?>> EVENT_DESIRIALIZERS = new TypeLiteral<EventDeserializer<?>>() {
20          private static final long serialVersionUID = 1L;
21      };
22  
23      @Override
24      public Handler create(Instance<Object> lookup, Parameters params) {
25          EventDeserializer<?> deserializer = lookup.select(EVENT_DESIRIALIZERS) //
26                  .handlesStream() //
27                  .filter(matchesType(params.get("eventType", Class.class))) //
28                  .findAny() //
29                  .orElseThrow() //
30                  .get();
31          Invoker<?, ?> invoker = params.get("invoker", Invoker.class);
32          return new SyntheticHandler<>(deserializer, invoker);
33      }
34  
35      private Predicate<Handle<?>> matchesType(Class<?> eventType) {
36          return h -> eventType.equals(h.getBean().getTypes().stream() //
37                  .filter(ParameterizedType.class::isInstance) //
38                  .map(ParameterizedType.class::cast) //
39                  .filter(t -> EventDeserializer.class.equals(t.getRawType())) //
40                  .findAny() //
41                  .orElseThrow() //
42                  .getActualTypeArguments()[0]);
43      }

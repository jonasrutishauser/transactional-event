26      void reset() {
27          for (String item : statements) {
28              try (Connection connection = datasource.getConnection();
29                   Statement statement = connection.createStatement()) {
30                  statement.execute(item);
31              } catch (SQLException e) {
32                  if (!item.startsWith("DROP")) {
33                      throw new IllegalStateException(e);
34                  }
35              }
36          }
37      }

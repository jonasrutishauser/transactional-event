5   import io.quarkus.runtime.annotations.ConfigPhase;
6   import io.quarkus.runtime.annotations.ConfigRoot;
7   import io.smallrye.config.ConfigMapping;
8   import io.smallrye.config.WithDefault;
9   
10  @ConfigMapping(prefix = "quarkus.transactional.event")
11  @ConfigRoot(phase = ConfigPhase.RUN_TIME)
12  public interface TransactionalEventConfiguration {
13      /**
14       * Time in ms to wait before next dispatching cycle is started if max concurrent
15       * dispatching threads are in use.
16       */
17      @WithDefault("" + Configuration.DEFAULT_ALL_IN_USE_INTERVAL)
18      int allInUseInterval();
19  
20      /**
21       * Max seconds to wait before a new check is made for unprocessed events.
22       */
23      @WithDefault("" + Configuration.DEFAULT_MAX_DISPATCHER_INTERVAL)
24      int maxDispatchInterval();
25  
26      /**
27       * Initial seconds to wait before a new check is made for unprocessed events.
28       */
29      @WithDefault("" + Configuration.DEFAULT_MAX_DISPATCHER_INTERVAL / 2)
30      int initialDispatchInterval();
31  
32      /**
33       * Number of unprocessed events to aquire in one transaction.
34       */
35      @WithDefault("" + Configuration.DEFAULT_MAX_CONCURRENT_DISPATCHING)
36      int maxAquire();
37  
38      /**
39       * Max number of events to process concurrently.
40       */
41      @WithDefault("" + Configuration.DEFAULT_MAX_CONCURRENT_DISPATCHING)
42      int maxConcurrentDispatching();
43  }

23  class QueryAdapterFactory {

25      private static final Logger LOGGER = LogManager.getLogger();
26  
27      private final QueryAdapter queryAdapter;
28  
29      public QueryAdapter getQueryAdapter() {
30          return queryAdapter;
31      }
32  
33      QueryAdapterFactory() {
34          queryAdapter = null;
35      }
36  
37      QueryAdapterFactory(DataSource dataSource) throws SQLException {
38          try (Connection connection = dataSource.getConnection()) {
39              queryAdapter = getQueryAdapter(connection);
40          }
41      }
42  
43      @Inject
44      QueryAdapterFactory(Instance<QueryAdapter> cdiInstance, @TransientReference @Events DataSource dataSource) {
45          if (cdiInstance.isUnsatisfied()) {
46              try (Connection connection = dataSource.getConnection()) {
47                  queryAdapter = getQueryAdapter(connection);
48              } catch (SQLException e) {
49                  throw new IllegalStateException(e);
50              }
51          } else {
52              queryAdapter = cdiInstance.get();
53          }
54      }

56      private static QueryAdapter getQueryAdapter(Connection connection) throws SQLException {
57          QueryAdapter queryAdapter;
58          String productName = connection.getMetaData().getDatabaseProductName();
59          if (productName.contains("Oracle")) {
60              queryAdapter = new OracleQueryAdapter();
61          } else if (productName.contains("MariaDB")) {
62              queryAdapter = new MariaDBQueryAdapter();
63          } else if (productName.contains("PostgreSQL") || productName.contains("MySQL")) {
64              queryAdapter = new LimitQueryAdapter();
65          } else {
66              Set<String> keywords = new HashSet<>(asList(connection.getMetaData().getSQLKeywords().split(",")));
67              if (keywords.contains("skip") && keywords.contains("locked")) {
68                  queryAdapter = new SkipLockedQueryAdapter();
69              } else {
70                  queryAdapter = new SimpleQueryAdapter();
71              }
72          }
73          LOGGER.debug(() -> "DB '" + productName + "' uses " + queryAdapter.getClass().getSimpleName());
74          return queryAdapter;
75      }
76  
77      private static class SimpleQueryAdapter implements QueryAdapter {
78          protected static final String LIMIT_EXPRESSION = "\\{LIMIT ([^}]+)\\}";
79  
80          @Override
81          public String fixLimits(String sql) {
82              return sql;
83          }
84  
85          @Override
86          public String addSkipLocked(String sql) {
87              return sql;
88          }
89      }
90  
91      private static class SkipLockedQueryAdapter extends SimpleQueryAdapter {
92          @Override
93          public String addSkipLocked(String sql) {
94              return sql.replaceAll(LIMIT_EXPRESSION, "").replace("FOR UPDATE", "FOR UPDATE SKIP LOCKED");
95          }
96      }
97  
98      private static class OracleQueryAdapter extends SkipLockedQueryAdapter {
99          @Override
100         public String addSkipLocked(String sql, int maxRows) {
101             return super.addSkipLocked(sql.replace("SELECT ", "SELECT /*+ FIRST_ROWS(" + maxRows + ") */ "));
102         }
103 
104         @Override
105         public String fixLimits(String sql) {
106             return sql.replaceAll(LIMIT_EXPRESSION, "AND rownum <= $1");
107         }
108     }
109 
110     private static class MariaDBQueryAdapter extends SimpleQueryAdapter {
111         @Override
112         public String fixLimits(String sql) {
113             return sql.replaceAll(LIMIT_EXPRESSION, "LIMIT $1");
114         }
115     }
116 
117     private static class LimitQueryAdapter extends SkipLockedQueryAdapter {
118         @Override
119         public String fixLimits(String sql) {
120             return sql.replaceAll(LIMIT_EXPRESSION, "LIMIT $1");
121         }
122     }

16  class QuarkusEventHandlers implements EventHandlers {
17  
18      private final Map<Class<?>, Class<? extends Handler>> handlerClass;
19  
20      QuarkusEventHandlers(Map<Class<?>, Class<? extends Handler>> handlerClass) {
21          this.handlerClass = handlerClass;
22      }
23  
24      @Override
25      public Optional<Class<? extends Handler>> getHandlerClassWithImplicitType(EventTypeResolver typeResolver,
26              String type) {
27          for (Entry<Class<?>, Class<? extends Handler>> handlerClassEntry : handlerClass.entrySet()) {
28              if (type.equals(typeResolver.resolve(handlerClassEntry.getKey()))) {
29                  return Optional.of(handlerClassEntry.getValue());
30              }
31          }
32          return Optional.empty();
33      }
34  
35      static class Creator implements SyntheticBeanCreator<EventHandlers> {
36  
37          @Override
38          public QuarkusEventHandlers create(Instance<Object> lookup, Parameters params) {
39              Class<?>[] types = params.get("types", Class[].class);
40              Class<?>[] beans = params.get("beans", Class[].class);
41              Map<Class<?>, Class<? extends Handler>> handlerClass = new HashMap<>();
42              for (int i = 0; i < types.length; i++) {
43                  handlerClass.put(types[i], beans[i].asSubclass(Handler.class));
44              }
45              return new QuarkusEventHandlers(handlerClass);
46          }
47  
48      }
